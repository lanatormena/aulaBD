CREATE TABLE cidade (
ID_CIDADE INTEGER(50) NOT NULL AUTO_INCREMENT,
nomeCidade VARCHAR(50) NOT NULL,
siglaCidade VARCHAR(3),
unidadeFederativa VARCHAR(2),
PRIMARY KEY (ID_CIDADE)
);
CREATE TABLE endereco (
	ID_ENDERECO INT NOT NULL AUTO_INCREMENT,
	logradouro VARCHAR(100),
	numeroPredial INTEGER,
	complementoLogradouro VARCHAR(30),
	cep INTEGER(8),
	ID_CIDADE INTEGER(50),
	PRIMARY KEY(ID_ENDERECO),
	FOREIGN KEY (ID_CIDADE) REFERENCES CIDADE(ID_CIDADE)
);
CREATE TABLE Cliente(
	ID_CLIENTE INTEGER(50) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	nomeCliente VARCHAR(50),
	cpfCliente INTEGER(11),
	ID_ENDERECO INTEGER(50),
	FOREIGN KEY (ID_ENDERECO) REFERENCES Endereco(ID_ENDERECO)
);
CREATE TABLE Usuario(
	ID_USUARIO INTEGER(50) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	nome VARCHAR(50),
	login VARCHAR(15),
	cpf INTEGER(11) NOT NULL,
	senha VARCHAR(50),
	ID_ENDERECO INTEGER(50),
	FOREIGN KEY (ID_ENDERECO) REFERENCES Endereco(ID_ENDERECO)
);
CREATE TABLE Recibo(
	ID_RECIBO INTEGER(50) NOT NULL AUTO_INCREMENT PRIMARY KEY,
	sequenciaRecibo INTEGER(50),
	valorRecibo DECIMAL(10,2),
	dataRecibo DATE,
	assinaturaRecibo TINYINT(1),
	referenciaRecibo VARCHAR(100),
	ID_CIDADE INTEGER(50),
	ID_USUARIO INTEGER(50),
	ID_CLIENTE INTEGER(50),
	FOREIGN KEY (ID_CIDADE) REFERENCES Cidade(ID_CIDADE),
	FOREIGN KEY (ID_USUARIO) REFERENCES Usuario(ID_USUARIO),
	FOREIGN KEY (ID_CLIENTE) REFERENCES Cliente(ID_CLIENTE)
);
